$entityName

$entityTable

$entityFields

Список полей сущности

$entityFieldsMultiLanguage

Список мультиязычных полей в сущности 

[
    'field1',
    'field2',
    ...
]
Пример в сущности 
    public $translatable = [
@foreach($entityFieldsMultiLanguage as $field)
        '{{ $field }}',
@endforeach
    ];

$entityFieldsDateTime

Список полей в сущности, которые являются датами 

[
    'field1',
    'field2',
    ...
]
Пример в сущности 
    protected $dates = [
@foreach($entityFieldsDateTime as $field)
        '{{ $field }}',
@endforeach
    ];

$entityFieldsCasts

Список полей в сущности для свойства сущности protected $casts = [... 

[
    'field1' => 'тип данных',
    'field2' => 'тип данных',
    ...
]
Пример в сущности 
    protected $casts = [
@foreach($entityFieldsCasts as $field => $type)
        '{{ $field }}' => '{{ $type }}',
@endforeach
    ];

$entityFieldCreatedAt

Включено поле "created_at" или нет

$entityFieldUpdatedAt

Включено поле "updated_at" или нет

$enableMultiLanguage

Присутствуют мультиязычные поля или нет

$useCommonForEntity

Список классов, которые необходимо использовать в разделе use модели

    ...

@foreach($useCommonForEntity as $use)

@endforeach
